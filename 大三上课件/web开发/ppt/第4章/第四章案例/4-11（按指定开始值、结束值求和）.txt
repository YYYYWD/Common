function sum(StartVal,EndVal){
var ArgNum=sum.arguments.length;
var i,s=0;
if (ArgNum==0) { StartVal=1;EndVal=1000;}
else if (ArgNum==1) EndVal=1000;
for (i=StartVal;i<=EndVal;i++) s+=i;
return s;
}
